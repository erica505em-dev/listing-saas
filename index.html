document.addEventListener("DOMContentLoaded", () => { const loginScreen = document.getElementById("loginScreen"); const loginBtn = document.getElementById("loginBtn"); const app = document.getElementById("app"); // LOGIN if (loginBtn) { loginBtn.addEventListener("click", () => { loginScreen.remove(); app.style.display = "flex"; showPage("dashboard"); }); } // SIDEBAR NAV const navButtons = document.querySelectorAll(".sidebar button"); navButtons.forEach(btn => { btn.addEventListener("click", () => { const page = btn.dataset.page; showPage(page); }); }); }); // PAGE ROUTER function showPage(id) { document.querySelectorAll(".page").forEach(p => p.style.display = "none"); const target = document.getElementById(id); if (target) target.style.display = "block"; }
